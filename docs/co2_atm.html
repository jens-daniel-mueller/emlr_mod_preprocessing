1 Read data

  • Data source: Globally averaged marine surface annual mean data from ESRL-NOAA
co2_atm <- tidync(
  paste0(path_atm_pCO2,
        "Atm_CO2_CESM-ETHZ_1_gr_1980-2018_v20211122.nc")
)

co2_atm <- co2_atm %>% 
  hyper_tibble()

co2_atm <- co2_atm %>%
  mutate(pCO2 = as.numeric(Atm_CO2),
         year = year(as.Date(time_ann, origin = "1980-01-01"))) %>%
  select(year, pCO2)

2 Time series

co2_atm %>% 
  ggplot(aes(year, pCO2)) +
  geom_path() +
  geom_point()
